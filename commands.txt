#to check version
git --version

#to check configuration
git config list

#to add user name
git config --global user.name 'username'

#to add user's email
git config --global user.email 'usermail'

#to initialize git
git init 

#to check the status 
git status

#to add the modification from local working directory to staging area
git add -A 

#to commit the changes to local repository
git commit -m 

#to show the modifications
git log

#command for adding local repository to remote repository
git push

#exact copy of remote repository to local repository
git clone

#to add changes in remote repository to lacal repository
git pull
